6   package kalalaji.vieheet;
7   
8   import java.io.File;
9   import javax.swing.ImageIcon;
10  import kalalaji.kalat.Kalat;
11  import saatila.logiikka.SaaNyt;
12  
13  /**
14   * Vakio luokka jossa määritetään vieheiden värikartta saan ja veden värin
15   * mukaan. Värikartta on haettu käytetyimmän ja parhaan värikartan mukaan
16   * kullekkin kalalajille.
17   *
18   * @see HAUKI: http://www.nilsmaster.fi/varikartta
19   */
20  public class Vieheet {
21  
22      private String[] vieheKuvat;
23      private final Kalat kalaNimi;
24      private ImageIcon kuva;
25  
26      /**
27       * Konstruktori.
28       *
29       * @param kalanimi Kalat kalanimi parametri.
30       */
31      public Vieheet(Kalat kalanimi) {
32          this.kalaNimi = kalanimi;
33      }
34  
35      private String[] haeVieheKuvat() {
36          File kansio = new File(getClass().getResource("/hauki").getFile());
37          vieheKuvat = kansio.list();
38          return vieheKuvat;
39      }
40  
41      /**
42       * Metodi.
43       *
44       * @param saanyt parametri.
45       * @param indeksi parametri.
46       * @return viehe kuva.
47       */
48      public ImageIcon naytaVieheKuva(SaaNyt saanyt, int indeksi) {
49          String[] kuvat = haeVieheKuvat();
50          String kuvanNimi = kuvat[indeksi];
51          //if (kuvanNimi.contains(saanyt.getPilvet().getPilvisyys().name()) || kuvanNimi.contains(saanyt.getSade().getSade().name())) {
52          this.kuva = new ImageIcon(getClass().getResource("/hauki/" + kuvanNimi));
53          return this.kuva;
54          //}
55          //return this.kuva;
56      }
57  }
