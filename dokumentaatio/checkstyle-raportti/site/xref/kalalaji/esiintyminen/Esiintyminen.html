54      private final String talvi;
55      private final String kevat;
56      private final String kesa;
57      private final String syksy;
58  
59      /**
60       * Luokan konstruktorissa määritetään vakio esiintymiset
61       *
62       * @param esTalvi on talvi esiintyminen String tyyppisenä
63       * @param esKevat on Kevät esiintyminen String tyyppisenä
64       * @param esKesa on Kesä esiintyminen String tyyppisenä
65       * @param esSyksy on Syys esiintyminen String tyyppisenä
66       */
67      Esiintyminen(String esTalvi, String esKevat, String esKesa, String esSyksy) {
68          this.talvi = esTalvi;
69          this.kevat = esKevat;
70          this.kesa = esKesa;
71          this.syksy = esSyksy;
72      }
73  
74      public String getTalviEsiintyminen() {
75          return this.talvi;
76      }
77  
78      public String getKevatEsiintyminen() {
79          return this.kevat;
80      }
81  
82      public String getKesaEsiintyminen() {
83          return this.kesa;
84      }
85  
86      public String getSyksyEsiintyminen() {
87          return this.syksy;
88      }
89  
90      /**
91       * Metodi vakion määrittelemän kalan esiintymisen hakuun sesongin mukaan.
92       *
93       * @param esiintyminen Esiintyminen vakio luokka.
94       * @param sesonkinyt SesonkiNyt olio.
95       * @return Vakio arvon esiintymisen sesongin mukaan String tyyppisenä.
96       */
97      public String esiintyySesonginMukaan(Esiintyminen esiintyminen, SesonkiNyt sesonkinyt) {
98          sesonkinyt = new SesonkiNyt();
99          switch (sesonkinyt.getSesonki()) {
100             case "TALVI":
101                 return esiintyminen.getTalviEsiintyminen();
102             case "KEVÄT":
103                 return esiintyminen.getKevatEsiintyminen();
104             case "KESÄ":
105                 return esiintyminen.getKesaEsiintyminen();
106             case "SYKSY":
107                 return esiintyminen.getSyksyEsiintyminen();
108             default:
109                 return "Jotain meni pieleen.";
110         }
111     }
