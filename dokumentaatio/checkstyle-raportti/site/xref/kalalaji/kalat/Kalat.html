6   package kalalaji.kalat;
7   
8   import aikapvm.SesonkiNyt;
9   import kalalaji.esiintyminen.*;
10  import kalalaji.kalastustapa.*;
11  
12  /**
13   * Vakio luokka joka luo yhden olion joka sisältää esiintymisen sekä
14   * kalastustavan sesongin mukaan.
15   *
16   */
17  public enum Kalat {
18      //Lisättynä vain hauki jotta ohjelma pysyisi siedettävässä mittakaavassa tässä vaiheessa.
19      //Ahven, Kuha, Taimen;
20      Hauki(HaukiEsiintyminen.HAUKI, HaukiKalastus.HAUKI);
21  
22      private Esiintyminen es;
23      private Kalastustapa kt;
24      private SesonkiNyt sn = new SesonkiNyt();
25  
26      /**
27       * Konstrukstori asettaa vakio arvolle sitä vastaavat parametrit.
28       *
29       * @param es on Esiintyminen rajapinta joka sisältää kaikki rajapinnan
30       * toteuttavat luokat
31       * @param kt on Kalastustapa rajapinta joka sisältää kaikki rajapinnan
32       * toteuttavat luokat
33       */
34      Kalat(Esiintyminen es, Kalastustapa kt) {
35          this.es = es;
36          this.kt = kt;
37      }
38  
39      /**
40       * Metodi palauttaa vakio arvona annetun kalan esiintymisen sesongin mukaan.
41       *
42       * @return esiintyminen sesongin mukaan String tyyppisenä.
43       */
44      public String esiintyySesonginMukaan() {
45          return es.esiintyySesonginMukaan(es, sn);
46      }
47  
48      /**
49       * Metodi hakee vakio kalaa koskevan kalastustavan.
50       *
51       * @return kalastus tapa sesongin mukaan String tyyppisenä.
52       */
53      public String kalastustapaSesonginMukaan() {
54          return kt.kalastusSesonginMukaan(kt, sn);
55      }
56  }
