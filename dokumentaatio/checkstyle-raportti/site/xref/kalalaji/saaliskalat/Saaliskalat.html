6   package kalalaji.saaliskalat;
7   
8   import aikapvm.SesonkiNyt;
9   
10  /**
11   * Saaliskalat vakio luokka jossa esiintyy sesongin mukaiset saaliskalat.
12   */
13  public enum Saaliskalat {
14  
15      HAUKI(//Talvi
16              "Made, Siika, Taimen, Ahven, Särki, Lahna, Säyne, Sulkava.",
17              //Kevät
18              "Ahven, Lahna, Säyne, Siika, Särki, Sulkava, Hauki, Kuha, Silakka.",
19              //Kesä
20              "Ahven, Säyne, Särki, Sulkava, Silakka, Hauki.",
21              //Syksy
22              "Ahven, Salakka, Särki, Lahna, Säyne, Silakka.");
23  
24      private final String talvi;
25      private final String kevat;
26      private final String kesa;
27      private final String syksy;
28  
29      /**
30       * Vakio luokan konstruktori.
31       *
32       * @param talvi on saaliskalojen nimet talvella.
33       * @param kevat on saaliskalojen nimet keväällä.
34       * @param kesa on saaliskalojen nimet kesällä.
35       * @param syksy on saaliskalojen nimet syksyllä.
36       */
37      Saaliskalat(String talvi, String kevat, String kesa, String syksy) {
38          this.talvi = talvi;
39          this.kevat = kevat;
40          this.kesa = kesa;
41          this.syksy = syksy;
42      }
43  
44      public String getTalviSaalisKalat() {
45          return this.talvi;
46      }
47  
48      public String getKevatSaalisKalat() {
49          return this.kevat;
50      }
51  
52      public String getKesaSaalisKalat() {
53          return this.kesa;
54      }
55  
56      public String getSyksySaalisKalat() {
57          return this.syksy;
58      }
59  
60      /**
61       *
62       * Metodi hakee saaliskalat sesongin mukaan.
63       *
64       * @param saaliskalat saaliskalat vakio luokka.
65       * @param sesonkinyt SesonkiNyt olio.
66       * @return vakio arvon saalis sesongin mukkan String tyyppisenä.
67       */
68      public String saalisSesonginMukaan(Saaliskalat saaliskalat, SesonkiNyt sesonkinyt) {
69          sesonkinyt = new SesonkiNyt();
70          switch (sesonkinyt.getSesonki()) {
71              case "TALVI":
72                  return saaliskalat.getTalviSaalisKalat();
73              case "KEVÄT":
74                  return saaliskalat.getKevatSaalisKalat();
75              case "KESÄ":
76                  return saaliskalat.getKesaSaalisKalat();
77              case "SYKSY":
78                  return saaliskalat.getSyksySaalisKalat();
79              default:
80                  return "Jotain meni pieleen.";
81          }
82      }
83  
84  }
