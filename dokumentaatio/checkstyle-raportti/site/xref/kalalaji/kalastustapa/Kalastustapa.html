6   package kalalaji.kalastustapa;
7   
8   import aikapvm.SesonkiNyt;
9   
10  /**
11   * Kalastustapa rajapinta jossa määritetään pakkauksessa kalalaji.kalastustapa
12   * olevien luokkien toteuttamat metodit. Rajapintaa käytetään myös selventämään
13   * pakkauksen luokkien olioiden toimintaa sekä perinnöllisyyttä, ja helpottamaan
14   * uusien lajien lissämistä ohjelmaan.
15   *
16   */
17  public interface Kalastustapa {
18  
19      /**
20       * Rajapinnan metodi joka palauttaa kalalajin talvi kalastus metodit.
21       *
22       * @return talvi kalastustapa String tyyppisenä.
23       */
24      public String getTalviKalastus();
25  
26      /**
27       * Rajapinnan metodi joka palauttaa kalalajin kevät kalastus metodit.
28       *
29       * @return kevät kalastustapa String tyyppisenä.
30       */
31      public String getKevatKalastus();
32  
33      /**
34       * Rajapinnan metodi joka palauttaa kalalajin kesä kalastus metodit.
35       *
36       * @return kesä kalastustapa String tyyppisenä.
37       */
38      public String getKesaKalastus();
39  
40      /**
41       * Rajapinnan metodi joka palauttaa kalalajin syksy kalastus metodit.
42       *
43       * @return syksy kalastustapa String tyyppisenä.
44       */
45      public String getSyksyKalastus();
46  
47      /**
48       * Metodi palauttaa tiedon string tyyppisenä missä haluttu laji esiintyy
49       * sesongin mukaan.
50       *
51       * @param kt on Kalastustapa rajapinnan ilmentymä joka sisältää kaikki
52       * rajapinnan toteuttavat luokat.
53       * @param sn SesonkiNyt olio joka määrittää ajankohtaisen sesongin.
54       * @return palauttaa kalalajin ennalta sesongin mukaan määritellyn
55       * kalastustavan kuvauksen String tyyppisenä.
56       */
57      public String kalastusSesonginMukaan(Kalastustapa kt, SesonkiNyt sn);
58  }
