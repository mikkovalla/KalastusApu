65      /**
66       * Metodi tarkistaa syötetyn vedenlämmön arvon. Asettaa oletuksen jos arvo
67       * ei ole oikea.
68       *
69       * @return vedenlämpö int tyyppisenä.
70       */
71      public int getVedenLampo() {
72          int lampo;
73          if (vedenLampoValitsin1.getText().isEmpty()) {
74              vedenLampoValitsin1.setText("14");
75              lampo = Integer.parseInt(vedenLampoValitsin1.getText());
76              return lampo;
77          }
78          if (Integer.parseInt(vedenLampoValitsin1.getText()) > 0 && Integer.parseInt(vedenLampoValitsin1.getText()) < 31) {
79              lampo = Integer.parseInt(vedenLampoValitsin1.getText());
80              return lampo;
81          }
82          lampo = 14;
83          return lampo;
84      }

88       *
89       * @return tuulennopeus int tyyppisenä.
90       */
91      public int getTuulenNopeus() {
92          if (tuulenNopeusValitsin.getText().isEmpty()) {
93              tuulenNopeusValitsin.setText("6");
94              tuulenNopeus = Integer.parseInt(tuulenNopeusValitsin.getText());
95              return tuulenNopeus;
96          }
97          if (Integer.parseInt(tuulenNopeusValitsin.getText()) > 0 && Integer.parseInt(tuulenNopeusValitsin.getText()) < 25) {
98              tuulenNopeus = Integer.parseInt(tuulenNopeusValitsin.getText());
99              return tuulenNopeus;
100         }
101         tuulenNopeus = 6;
102         return tuulenNopeus;
103     }
