6   package aikapvm;
7   
8   /**
9    *
10   * Kuukaudet Enum luokka tarjoaa Kuukaudet vakio arvoina, sekä kuukauden numeron
11   * ja Sesongin nimen. Tämä selventää huomattavasti kuukauden hakua ja sesongin
12   * hakua kuukauden perusteella.
13   */
14  public enum Kuukaudet {
15  
16      TAMMIKUU(1, "TALVI"), HELMIKUU(2, "TALVI"), MAALISKUU(3, "KEVÄT"),
17      HUHTIKUU(4, "KEVÄT"), TOUKOKUU(5, "KEVÄT"), KESÄKUU(6, "KESÄ"), HEINÄKUU(7, "KESÄ"),
18      ELOKUU(8, "KESÄ"), SYYSKUU(9, "SYKSY"), LOKAKUU(10, "SYKSY"), MARRASKUU(11, "SYKSY"), JOULUKUU(12, "TALVI");
19  
20      private int kuukausiNumero;
21      private String sesonki;
22  
23      /**
24       *
25       * @param kn on kuukauden numero joka asettaa kullekkin kuukauden arvolle
26       * tätä vastaavan numerollisen arvon.
27       * @param sesonki on String tyyppinen ja kertoo mikä sesonki vallitsee
28       * tietyn kuukauden aikana.
29       *
30       * Kuukaudet Enum luokan konstruktori.
31       */
32      Kuukaudet(int kn, String sesonki) {
33          this.kuukausiNumero = kn;
34          this.sesonki = sesonki;
35      }
36  
37      /**
38       *
39       * Metodi palauttaa kuukauden numeron int tyyppisenä.
40       *
41       * @return Kuukauden numerona int tyyppisenä.
42       */
43      public int getKuukausiNumero() {
44          return this.kuukausiNumero;
45      }
46  
47      /**
48       *
49       * Metodi palauttaa kuukauden sesongin String muodossa.
50       *
51       * @return Kuukauden sesonki String tyyppisenä.
52       */
53      public String getKuukaudenSesonki() {
54          return this.sesonki;
55      }
56  }
